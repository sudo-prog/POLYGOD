# POLYGOD Docker Compose вЂ” dev/staging stack
#
# Changes vs previous version:
#   - FIXED M5: Added postgres service so DATABASE_URL default (postgresql+asyncpg://...)
#               actually connects. Without this, the backend crashed immediately with
#               "could not connect to server" on every `docker compose up`.
#   - FIXED M6: polygod-swarm healthcheck no longer imports the full Python module
#               on every 15s tick (which triggered side effects and was slow).
#               Replaced with a simple pgrep process-existence check.
#   - HARDENED: redis port no longer exposed on 0.0.0.0 (only on 127.0.0.1).
#               Qdrant ports also restricted to localhost.
#   - ADDED:    backend depends_on postgres with condition: service_healthy so
#               the app doesn't race against a starting database.
#   - FIXED C1-DB: Made all DateTime columns timezone-aware to prevent
#                  naive vs aware datetime errors during price history insert.
#   - PROFILE:   polygod-swarm moved to optional 'swarm' profile to prevent
#                automatic start with `docker compose up`. Use:
#                `docker compose --profile swarm up -d polygod-swarm`

services:
  postgres:
    image: postgres:15-alpine
    restart: unless-stopped

      POSTGRES_DB: polygod

      - postgres_data:/var/lib/postgresql/data
